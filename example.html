>< associative of add

(eva

 (+ natural (-> type)
    zero (-> natural)
    succ (natural -> natural))

 (~ add (natural natural -> natural)
    (:m zero -> :m)
    (:m :n succ -> :m :n add succ))

 (~ mul (natural natural -> natural)
    (:m zero -> zero)
    (:m :n succ -> :m :n mul :m add))

 (+ eq ({:t : type} :t :t -> type)
    refl ({:t : type} {:d : :t} -> :d :d eq))

 (~ eq/test1
    (-> :z zero succ succ eq
        :y :z eq
        :y zero succ zero succ add eq)
    (-> refl refl refl))

 (~ cong
    ({:t1 :t2 : type} {:x :y : :t1}
     :x :y eq (:f : (:t1 -> :t2)) ->
     :x :f apply :y :f apply eq)
    (refl :f -> refl))

 (~ associative
    ((:x :y :z : natural) ->
     :z :x :y add add
     :z :y add :x add eq)
    (:x :y zero -> refl)
    (:x :y :z succ -> :x :y :z associative (:n -> :n succ) cong)
    ))

natural-induction

(eva

 (+ natural (-> type)
    zero (-> natural)
    succ (natural -> natural))

 (~ add (natural natural -> natural)
    (:m zero -> :m)
    (:m :n succ -> :m :n add succ))

 (~ mul (natural natural -> natural)
    (:m zero -> zero)
    (:m :n succ -> :m :n mul :m add))

 (~ natural-induction

    ((:p : (natural -> type))
     zero :p apply
     ((:k : natural) :k :p apply -> :k succ :p apply)
     (:x : natural) -> :x :p apply)

    (:p :p/z :p/s zero -> :p/z)
    (:p :p/z :p/s :k succ ->
        :k
        :p :p/z :p/s :k natural-induction
        :p/s apply)))
